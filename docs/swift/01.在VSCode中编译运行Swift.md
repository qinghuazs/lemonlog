

```bash
# 切换开发者目录
sudo xcode-select -s /Applications/Xcode.app/Contents/Developer

# 编译项目 demo 是工程名称
xcodebuild -project demo.xcodeproj -scheme demo -destination 'platform=iOS Simulator,name=iPhone 16' build

# 启动模拟器并安装运行应用
xcrun simctl boot "iPhone 16"
xcrun simctl install "iPhone 16" build/Release-iphonesimulator/demo.app
xcrun simctl launch "iPhone 16" com.moon.cloud.demo
```
